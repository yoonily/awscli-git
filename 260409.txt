1. SageMaker 인프라 구성(Terraform)
노트북 인스턴스와 이를 위한 보안 그룹, iam 역할 생성
# 1. SageMaker 실행을 위한 IAM Role
resource "aws_iam_role" "sagemaker_role" {
  name = "sagemaker-execution-role"

  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Action = "sts:AssumeRole"
      Effect = "Allow"
      Principal = { Service = "sagemaker.amazonaws.com" }
    }]
  })
}

# SageMaker 기본 권한 및 Bedrock 접근 권한 추가 (GenAI 활용 대비)
resource "aws_iam_role_policy_attachment" "sagemaker_full_access" {
  role       = aws_iam_role.sagemaker_role.name
  policy_arn = "arn:aws:iam::aws:policy/AmazonSageMakerFullAccess"
}

resource "aws_iam_role_policy_attachment" "bedrock_access" {
  role       = aws_iam_role.sagemaker_role.name
  policy_arn = "arn:aws:iam::aws:policy/AmazonBedrockFullAccess"
}

2. Bedrock 기반 RAG 시스템 구성을 위한 권한 배포
PostgreSQL(AgensSQL) 벡터 데이터베이스 사용 생성형 AI 시스템 구축 Bedrock 호출 권
# Bedrock 호출을 위한 전용 IAM Policy
resource "aws_iam_policy" "bedrock_invoke" {
  name        = "BedrockInvocationPolicy"
  description = "Policy to allow invoking Bedrock models"

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Action   = "bedrock:InvokeModel"
        Effect   = "Allow"
        Resource = "arn:aws:bedrock:ap-northeast-2::foundation-model/*"
      }
    ]
  })
}
